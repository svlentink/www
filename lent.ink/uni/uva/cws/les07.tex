\section{2.5.4}

Given that (source: Elementary Probability for Applications, p13):
\\
\begin{equation}
P (A) = P (A \cap B) + P (A \cap B^c )
\end{equation}
\begin{equation}
P (B) = P (B \cap A) + P (B \cap A^c )
\end{equation}
\begin{equation}
P (A) + P (B) - P (A \cap B) =
P (A \cap B) + P (A \cap B^c )
+ P (B \cap A^c ) = P (A \cup B)
\end{equation}

From this we get

\begin{equation}
P (A \cap B) = -P (A \cup B) + P (A) + P (B)
\end{equation}
\begin{equation}
P(A^c \cup B^c) = P((A \cap B)^c) %= P(A^c) + P(B^c) -
\end{equation}
\begin{equation}
P (A^c \cap B) =
P (B) - P (A \cap B)
\end{equation}
